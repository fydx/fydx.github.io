LCS最长公共子串

LCS（最长连续公共子串）算法

假设我们有两个字符串 ABCDMARK1234 ， MARKDOWN34

那么最长公共子串就是MARK

我们建立假设原来的子串分别为A，B，我们建立一个新的数组C，去记录最长的连续子串的位置。

递推公式：
C[i,j] = 0 (A[i]不等于B[j])
= 1 (A[i]等于B[i]且i或j小于1)
= C[i-1,j-1] + 1 (A[i]等于B[j])

使用m，n记录C数组中最大的位置

public class Main {

public static String A = "ABCDMARK1234" ;
public static String B = "MARKDOWN" ;
public static int[][] C ;
public static int m;
public static int n;
public static int maxValue = 0;
public static void main(String[] args)
{
    /* 创建二维数组C */
    C = new int[A.length()][B.length()];
    /* i,j循环 */
    for (int i = 0 ; i<A.length();i++)
        for (int j =0 ; j<B.length(); j++)
        {
            if (A.charAt(i) == B.charAt(j)) {
                /* 如果i或者j小于1，那么当前的值直接置为1 */
                if (i<1 || j <1)
                    C[i][j] = 1;
                else{
                    C[i][j] = C[i - 1][j - 1] + 1;
                }

                if (C[i][j] > maxValue)
                {
                    maxValue = C[i][j];
                    m = i;
                    n = j;
                }
            }
            else
                C[i][j] = 0;

        }
    /* 将m和n指向匹配第一个字符的位置 */
    m = m - maxValue + 1 ;
    n = n - maxValue + 1 ;
    /* 循环输出 */
    for (int i=0;i<maxValue;i++)
    {
        System.out.print(A.charAt(m));
        m++;
    }
}
}
