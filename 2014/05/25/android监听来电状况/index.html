android监听来电状况

建立一个BoardCastReceiver进行监听

/**
* Created by libin on 14-5-24.
*/
public class ComingCallReceiver extends BroadcastReceiver {

private final String TAG = "PhoneState";
private static String phoneNumber = null;
@Override
public void onReceive(Context context, Intent intent) {
    String action = intent.getAction();
    if(intent.getAction().equals(Intent.ACTION_NEW_OUTGOING_CALL)){
        //拨打电话的状况
    }else{
        //android没有接收来电的action，所以，不是拨打电话，就是接收来电
        doReceivePhone(context, intent);
    }

}

/**
 * 处理来电
 * @param context
 * @param intent
 */
public void doReceivePhone(Context context, Intent intent) {

    TelephonyManager telephony =
            (TelephonyManager)context.getSystemService(Context.TELEPHONY_SERVICE);
    int state = telephony.getCallState();
    switch(state){
        case TelephonyManager.CALL_STATE_RINGING:
            Log.i(TAG, "来电 ："+phoneNumber);
            /* 这里只有在来电的时候才能获取到PhoneNumber，其他的时候用这种方式获取的是null，
                所以PhoneNumber使用一个静态变量来保存 */
            phoneNumber = intent.getStringExtra(
                    TelephonyManager.EXTRA_INCOMING_NUMBER);
            break;
        case TelephonyManager.CALL_STATE_IDLE:
            Log.i(TAG, "挂断 ："+phoneNumber);
            break;
        case TelephonyManager.CALL_STATE_OFFHOOK:
            Log.i(TAG, "通话 ："+phoneNumber);
            break;
    }
}
}

当然，我们还需要找个地方（Activity中或Service中）注册一下Receiver

 public void registerIt() {
    Log.i("PhoneState", "registerIt");
    ComingCallReceiver  comingCallReceiver= new ComingCallReceiver();
    IntentFilter intentFilter = new IntentFilter();
    intentFilter.addAction(TelephonyManager.ACTION_PHONE_STATE_CHANGED);
    intentFilter.setPriority(Integer.MAX_VALUE);
    registerReceiver(comingCallReceiver, intentFilter);
}
